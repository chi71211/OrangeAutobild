AutoBild 爬蟲系統 v11.0 流程圖
================================

【完整流程圖】

```mermaid
flowchart LR
    %% --------------------------------------------------
    %% 節點樣式設定 (模擬參考圖片的專業配色)
    %% --------------------------------------------------
    classDef default fill:#f8f9fa,stroke:#ced4da,stroke-width:1px,color:#212529
    classDef decision fill:#ffffff,stroke:#6c757d,stroke-width:2px,color:#212529,shape:diamond
    classDef green_box fill:#d1e7dd,stroke:#198754,stroke-width:1px,color:#0f5132
    classDef red_box fill:#f8d7da,stroke:#dc3545,stroke-width:1px,color:#842029
    classDef blue_box fill:#cff4fc,stroke:#0dcaf0,stroke-width:1px,color:#055160
    classDef db_box fill:#fff3cd,stroke:#ffc107,stroke-width:1px,color:#664d03
    
    %% 區塊背景設定 (模擬圖片中的深灰色面板)
    style Phase1 fill:#2b2b2b,stroke:#555,stroke-width:2px,color:#fff
    style Phase2 fill:#2b2b2b,stroke:#555,stroke-width:2px,color:#fff
    style Phase3 fill:#2b2b2b,stroke:#555,stroke-width:2px,color:#fff
    style Phase4 fill:#2b2b2b,stroke:#555,stroke-width:2px,color:#fff
    style Phase5 fill:#2b2b2b,stroke:#555,stroke-width:2px,color:#fff
    style Phase6 fill:#2b2b2b,stroke:#555,stroke-width:2px,color:#fff

    %% --------------------------------------------------
    %% ① 啟動前進度與週期檢查
    %% --------------------------------------------------
    subgraph Phase1 ["① 啟動前進度檢查"]
        direction TB
        Start([系統啟動]) --> ReadDB[("讀取 SQLite<br>system_metadata")]:::db_box
        ReadDB --> CheckCycle{"上次執行狀態<br>(cycle_start)"}:::decision
        CheckCycle -- "< 7 天<br>(未完成)" --> Resume["接續執行<br>載入已完成清單"]:::green_box
        CheckCycle -- "> 7 天<br>(或已完成)" --> Reset["清空 progress 表<br>開始全新週期"]:::green_box
        Resume --> Launch["啟動無頭 Chromium<br>headless=True"]:::red_box
        Reset --> Launch
    end

    %% --------------------------------------------------
    %% ⑤ 防護與錯誤恢復 (對應圖片放置於側邊/底部的中斷邏輯)
    %% --------------------------------------------------
    subgraph Phase5 ["⑤ 防護與錯誤恢復 (全域攔截)"]
        direction TB
        SignalCatch{"接收到 Timeout<br>或 Ctrl+C / SIGTERM"}:::decision
        SignalCatch -- "是" --> TriggerSafe["緊急安全存檔<br>標記 is_interrupted = True"]:::red_box
        TriggerSafe --> ForceSave[("強制寫入殘留暫存<br>Flush Batch")]:::db_box
    end

    %% --------------------------------------------------
    %% ② 瀏覽器自動化與導航
    %% --------------------------------------------------
    subgraph Phase2 ["② 瀏覽器自動化與導航"]
        direction TB
        NavURL["進入目標網址<br>marken-modelle"] --> Cookie{"出現 Cookie<br>彈窗?"}:::decision
        Cookie -- "是" --> Bypass["點擊 Alle akzeptieren<br>突破阻擋"]:::red_box
        Cookie -- "否" --> ScrollBrand
        Bypass --> ScrollBrand["向下滾動 3000px<br>收集全站品牌連結"]:::blue_box
        ScrollBrand --> GotoBrand[進入特定品牌頁面]
        GotoBrand --> ExpandModels["點擊 'ALLE MODELLE'<br>解鎖隱藏車系"]:::red_box
    end

    %% --------------------------------------------------
    %% ③ 動態萃取與攔截
    %% --------------------------------------------------
    subgraph Phase3 ["③ 動態渲染與深度萃取"]
        direction TB
        CheckVar{"檢查頁面是否<br>有規格變體?"}:::decision
        CheckVar -- "有變體" --> ClickAllVar["操作 DOM 元素<br>暴力展開燃料變體"]:::red_box
        CheckVar -- "單一規格" --> ParseJSON
        ClickAllVar --> ParseJSON["解析 #vike_pageContext<br>取得底層 JSON"]:::blue_box
        ParseJSON --> MapData["提取: 車身/年份/馬力"]:::blue_box
        MapData --> APICatch["監聽 context.on('response')<br>攔截 API 提取 HSN/TSN"]:::green_box
        APICatch --> Fallback{"API 攔截<br>是否成功?"}:::decision
        Fallback -- "否" --> DOMCatch["備用方案：解析<br>DOM Overlay 結構"]:::blue_box
        Fallback -- "是" --> MapDone([萃取完成])
        DOMCatch --> MapDone
    end

    %% --------------------------------------------------
    %% ④ 資料清洗與翻譯
    %% --------------------------------------------------
    subgraph Phase4 ["④ 資料清洗與翻譯"]
        direction TB
        TranslateBody["匹配 body_map<br>(e.g. fliessheck ➔ 掀背車)"]:::green_box
        TranslateBody --> TranslateFuel["匹配 fuel_map<br>(e.g. benzin ➔ 汽油)"]:::green_box
    end

    %% --------------------------------------------------
    %% ⑥ 斷點記憶與週期管理
    %% --------------------------------------------------
    subgraph Phase6 ["⑥ 斷點記憶與匯出管理"]
        direction TB
        CheckBatch{"暫存佇列滿 50 筆?"}:::decision
        CheckBatch -- "是" --> BatchDB[("寫入 car_catalog<br>(REPLACE INTO)")]:::db_box
        CheckBatch -- "否" --> SkipDB([繼續抓取下一筆])
        BatchDB --> UpdateProg["更新 model_progress<br>紀錄已完成車系"]:::green_box
        UpdateProg --> CheckBrandEnd{"品牌是否<br>掃描完畢?"}:::decision
        CheckBrandEnd -- "是" --> Export["透過 SQL View<br>匯出品牌 CSV 報表"]:::db_box
        CheckBrandEnd -- "否" --> NextModel([前往下個車系])
        Export --> CheckAllEnd{"全站掃描<br>是否完畢?"}:::decision
        CheckAllEnd -- "是" --> MarkDone["更新 system_metadata<br>標記狀態為 completed"]:::green_box
        MarkDone --> SysExit([完美退出])
    end

    %% --------------------------------------------------
    %% 跨區塊動線連結 (主流程)
    %% --------------------------------------------------
    Launch ==> NavURL
    ExpandModels ==> CheckVar
    MapDone ==> TranslateBody
    TranslateFuel ==> CheckBatch

    %% --------------------------------------------------
    %% 跨區塊動線連結 (錯誤攔截)
    %% --------------------------------------------------
    Launch -. "超時/中斷" .-> SignalCatch
    NavURL -. "超時/中斷" .-> SignalCatch
    CheckVar -. "超時/中斷" .-> SignalCatch
    ForceSave ==> SysExit
```
